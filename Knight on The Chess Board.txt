# -Python-
# 8x8 크기의 체스판에서 특정 위치에 있는 나이트가 움직일 수 있는 경우의 수를 구하는 메소드
# 입력 : 나이트의 위치(pos)('문자+숫자' ex)'B5')		출력 : 나이트가 움직일 수 있는 경우의 수(solution)

# 나이트 위치에 따른 경우의 수가 체스판 중심을 기준으로 점대칭, 상하 선대칭, 좌우 선대칭 구조를 이루고 있다
# ㄴ즉, 8등분 했을 때 모두 같은 모양과 속성을 지니고 있다

def solution(pos):
	size = 8
	# 체스판의 크기
	
	x = ord(pos[0])-64
	# x는 가로 좌표, 문자로 입력받기 때문에 숫자로 변환 (A=1, B=2, ...)
	if x>size/2:
	# 중심 세로선을 기준으로 오른 쪽에 위치할 경우
		x = size+1-x
		# 왼 쪽으로 대칭 이동
		
	y = int(pos[1])
	# y는 세로 좌표, 문자열 두번 째 자리의 문자를 정수로 치환하여 가져옴 ('3' -> 3)
	if y>size/2:
	# 중심 가로선을 기준으로 위 쪽에 위치할 경우
		y = size+1-y
		# 밑으로 대칭 이동
	
	# 좌하단으로 이동 된 상태
	if(x>y):
	# 중심 대각선을 기준으로 우하단에 위치할 경우
		temp=x
		x=y
		y=temp
		# 좌상단으로 대칭 이동
	
	# 경우의 수가 10개 밖에 되지 않으므로 쭉 나열
	if x==3 or x==4:
		answer=8
		
	elif x==2:
		if y==2:
			answer=4
		else:
			answer=6
			
	else:
		if y==2:
			answer = 3
		else:
			answer = 2
	# 체스판의 크기가 아주 커 경우의 수가 늘어나면 어떻게 해야할까?
	
	return answer
